private static void rodCutting(int[] prices, int n, int[] revenue, int[] cutIndices) {        revenue[0] = 0;        for (int i = 1; i <= n; i++) {            int maxRevenue = Integer.MIN_VALUE;            for (int j = 1; j <= i; j++) {                if (maxRevenue < prices[j - 1] + revenue[i - j]) {                    maxRevenue = prices[j - 1] + revenue[i - j];                    cutIndices[i] = j;                }            }            revenue[i] = maxRevenue;        }    }    private static void printCutIndices(int n, int[] cutIndices) {        while (n > 0) {            System.out.print(cutIndices[n] + " ");            n = n - cutIndices[n];        }        System.out.println();    }
